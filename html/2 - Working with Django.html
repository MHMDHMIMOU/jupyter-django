from django.template import Context, Template
from IPython.display import HTML
In [17]:
top_venues = (
    gigs.values('venue__name', 'venue__city')
    .annotate(gig__count=Count('*'))
    .order_by('-gig__count')
    [:10]
)

template = Template("""
<table>
    <tr>
        <th>Venue</th>
        <th>City</th>
        <th>Gigs</th>
    </tr>
    {% for v in venues %}
    <tr>
        <td>{{v.venue__name}}</td>
        <td>{{v.venue__city}}</td>
        <td>{{v.gig__count}}</td>
    </tr>
    {% endfor %}
</table>
""")

context = Context(
    {'venues': top_venues}
)

HTML(template.render(context))
